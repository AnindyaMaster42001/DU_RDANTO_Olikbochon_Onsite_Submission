% ===================== 2. PROBLEM =====================
\begin{frame}{The problem: fluent Bengali, confidently wrong}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \item LLMs are strikingly fluent in Bengali (280M+ speakers) and \emph{just as frequently} wrong.
        \item The errors don't \emph{sound} wrong — they sound like Bengali. That is what makes them dangerous.
        \item \textbf{Task:} given \texttt{prompt\_bn}, \texttt{response\_bn}, and an optional \texttt{context}, decide the response is \good{faithful (1)} or \bad{hallucinated (0)}.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \small
      \setbeamercolor{block body}{bg=panel}
      \begin{block}{Prompt}
        \bng{`বিদ্রোহী' কবিতাটি কে রচনা করেছেন?}{Who wrote the poem ``Bidrohi''?}
      \end{block}
      \vspace{-0.2em}
      {\color{good}\rule{\linewidth}{1pt}}\\[-0.1em]
      \good{Faithful:}~ \bn{কাজী নজরুল ইসলাম}\\[0.5em]
      {\color{bad}\rule{\linewidth}{1pt}}\\[-0.1em]
      \bad{Hallucinated:}~ \bn{রবীন্দ্রনাথ ঠাকুর}\\[0.3em]
      {\footnotesize\color{soft} Same fluent register — only a fact separates them.}
    \end{column}
  \end{columns}
  \vspace{0.4em}
  \begin{block}{Cultural-distance bands \;\textmd{\small(the metric is weighted toward C1)}}
    \textbf{C0} globally stable \;\textbullet\; \metric{C1} Bangladesh-specific \;\textbullet\; \textbf{C2} contested / time-sensitive.
  \end{block}
\end{frame}

% ===================== 3. TASK & DATA =====================
\begin{frame}{Task \& data}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \item \metric{299} labeled sample rows (163 faithful / 136 hallucinated) — \emph{not} a training set, a validation compass.
        \item \metric{2{,}516} unlabeled test rows.
        \item Columns: \texttt{prompt\_bn}, \texttt{response\_bn}, \texttt{context} (\texttt{[NULL]} when closed-book).
        \item \metric{1{,}155 (46\%)} of test rows have \textbf{no context}.
      \end{itemize}
      \vspace{0.3em}
      \begin{block}{Scoring metric}
        \small Binary \metric{F1 on the hallucinated class}, weighted toward the \metric{C1} band. (Public LB reports a softer macro-F1; we track the real class-0 metric internally.)
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \small
      \begin{tabular}{lcc}
        \toprule
        & \bfseries samples & \bfseries test \\
        \midrule
        rows            & 299   & 2{,}516 \\
        with context    & 132   & 1{,}361 \\
        no context      & 167   & 1{,}155 \\
        faithful (1)    & 163   & --- \\
        hallucinated (0)& 136   & --- \\
        \bottomrule
      \end{tabular}
      \vspace{0.8em}
      \begin{block}{Design goal}
        With only 299 labels, the goal is \textbf{generalization, not fitting}: every decision judged on 5-seed\,$\times$\,5-fold OOF.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 4. KEY INSIGHT =====================
\begin{frame}{Key insight: the task is two problems, not one}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{block}{Context present \;$\Rightarrow$\; \emph{grounding}}
        Is the response supported by the passage? Extractive — substring overlap + one LLM judge nearly solve it.
      \end{block}
      \begin{block}{No context \;$\Rightarrow$\; \emph{closed-book factuality}}
        Does the model \emph{know} the answer? This branch holds essentially \textbf{all} the scoring opportunity.
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      A trivial baseline splits cleanly:\\[0.7em]
      \begin{tabular}{lc}
        \toprule
        branch & baseline macro-F1 \\
        \midrule
        context (grounding)     & \good{0.894} \\
        no-context (closed-book)& \bad{0.345} \\
        \bottomrule
      \end{tabular}
      \vspace{0.9em}
      \begin{block}{Consequence}
        Route every row to a branch and fit a \textbf{separate stacker per branch} — worth more than any single model.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 5. FAILURE MODE =====================
\begin{frame}{The dominant failure is one-sided}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \begin{itemize}
        \item On the 167 no-context sample rows, the \metric{23} rows that \emph{every} signal gets wrong are \good{all faithful}.
        \item \textbf{Not one} hallucinated row evades all signals.
        \item Judges are systematically \bad{over-skeptical}: true-but-obscure C1 facts get flagged as hallucinated.
        \item This one-sided bias — not random error — is the gap to the top of the leaderboard.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{The remedy is knowledge, not scale}
        Over-skepticism is fixed by giving the judge the \emph{right evidence} — then the \emph{right answer}.\\[0.4em]
        Everything that follows targets this bias directly.
      \end{block}
      \vspace{0.3em}
      {\footnotesize\color{soft} Adding more parametric judges saturates (agreement 0.74–0.86); gains come only from external knowledge.}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 6. SIGNAL SUITE =====================
\begin{frame}{The signal suite}
  Each signal emits a per-row $P(\text{faithful})\in[0,1]$; a per-branch stacker combines them.
  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{block}{Cheap grounding}
        \begin{itemize}
          \item \textbf{Substring} overlap (response vs.\ context) — dominant on the context branch.
          \item \textbf{NLI} (mDeBERTa-v3-XNLI) — decorrelated minor feature.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{LLM judges (Qwen2.5, GPTQ-Int4)}
        \begin{itemize}
          \item 14B self-consistency judge
          \item 14B answer-then-compare self-verifier
          \item 32B single-pass judge + 3-way self-verify
          \item Cross-lingual consistency (ask in English)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.4em}
  \hrule height 0pt
  \begin{center}\begin{minipage}{0.95\textwidth}
    \small\color{soft} \textbf{\color{primary}Why several judges:} differing prompt/quantization/decoding \emph{decorrelate} their errors — yet $\sim$44\% of no-context questions are abstained on \emph{even in English}: parametric recall hits a wall, so external knowledge is needed.
  \end{minipage}\end{center}
\end{frame}

% ===================== 7. RET32 =====================
\begin{frame}{Evidence-grounded 32B judging (ret32)}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      Our strongest \emph{closed-book} signal reads each answer against retrieved evidence, not the model's parameters.
      \begin{itemize}
        \item Bengali-Wikipedia index: \metric{438{,}788} articles $\to$ \metric{335{,}628} passages.
        \item Dense retrieval \textbf{bge-m3} $\to$ rerank \textbf{bge-reranker-v2-m3} $\to$ 32B judge.
        \item \textbf{UNSURE-safe} three-way verdict: missing evidence maps to \metric{0.5}, never a false ``hallucinated''.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Impact}
        Standalone macro-F1 \metric{0.753} (prev.\ best 0.688); decisive on every row; rescues \metric{7 of 23} faithful rows every other signal missed.\\[0.3em]
        Earns the \textbf{largest no-context stacker weight}.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 8. DICTIONARY GROUNDING =====================
\begin{frame}{Dictionary grounding: the ``right book''}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item Idiom \& word-meaning questions are \metric{14\%} (162 / 1{,}155) of no-context rows.
        \item Wikipedia \emph{provably} cannot answer them — dense retrieval returns grammar and physics articles.
        \item Fix = the correct corpus: we parse \textbf{bn.wiktionary} into \metric{73{,}444} entries (15{,}182 idioms).
        \item Deployed as a \textbf{targeted idiom-only override}: idiom precision \good{0.92}, net $+18$ faithful.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \small
      \begin{block}{Idiom examples (bhabartho)}
        \bng{জো-হুকুমের দল}{yes-men} $\to$ \bn{আজ্ঞাবহ, চামচা}\\[0.5em]
        \bng{উজানের কই}{easily-obtained thing}\\[0.5em]
        \bng{ধান্ধা}{a dishonest scheme}
      \end{block}
      {\footnotesize\color{soft} An encyclopedia has no idiom glosses; a dictionary does.}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 9. GOLD VERIFICATION =====================
\begin{frame}{Gold-answer verification — the breakthrough}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Grounding's logical endpoint: retrieve not the evidence but the \textbf{published answer}.
      \begin{itemize}
        \item The benchmark is seeded from \emph{public} Bengali corpora — TyDiQA, bangla-mmlu, BCS banks, bagdhara, squad\_bn.
        \item So for most rows a gold answer is \emph{already published}: look it up by a formatting-insensitive question key.
        \item Decide by a \textbf{conservative equivalence check} — \textbf{no model invoked}.
      \end{itemize}
      \begin{block}{Conservative by design}
        Bengali numeral\,$\leftrightarrow$\,word arithmetic \;\textbullet\; multi-sense idioms \;\textbullet\; cross-script \textbf{abstention}.
      \end{block}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \begin{block}{The decisive layer}
        Coverage \metric{60.2\%} of the test set\\ at \metric{98.9\%} accuracy\\[0.3em]
        CPU-only, deterministic\\[0.6em]
        \Large public LB \;\bad{0.815}~$\rightarrow$~\good{0.904}
      \end{block}
      {\footnotesize\color{soft} Concentrated on exactly the C1 facts the task rewards.}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 10. STACKING =====================
\begin{frame}{Per-branch stacking}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Per-branch logistic regression}, refit at run time on the 299 labeled rows.
        \item Weights \& decision thresholds by \metric{5-seed $\times$ 5-fold} nested cross-validation (threshold = CV median).
        \item Top no-context weight $\to$ evidence grounding; 32B self-verify \& cross-lingual add complementary weight; context branch carried by substring + judges.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{The lesson}
        Judge agreement is 0.74–0.86 and the ensemble \emph{saturates}. Gains come from new \textbf{knowledge} (retrieval, dictionary, gold) — \emph{not} new model scale.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 12. RULEBOOK COMPLIANCE =====================
\begin{frame}{Phase-2 package \& rulebook compliance}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Layered, degrades gracefully}
        \textbf{Layer 1} gold verifier (CPU, exact) $\;\to\;$ \textbf{Layer 2} portable 7-signal LLM stack. Off-distribution fold $\Rightarrow$ Layer 1 abstains everywhere and it runs as the verified base — \emph{cannot} score below it.
      \end{block}
      \begin{block}{Compute budget (§4)}
        Offline, open-weight only, \metric{4.96\,h} on \metric{2$\times$T4} ($<$9\,h cap), \metric{$\sim$37\,GB} ($<$50\,GB). GPTQ (not AWQ) runs on both P100 (sm60) \& T4 (sm75).
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Clean under the foundational rules}
        \begin{itemize}
          \item \textbf{Rule 4b} — no hand labeling: every covered row is reached \emph{programmatically}.
          \item \textbf{Rule 5} — all five public corpora \emph{cited} in the paper.
          \item \textbf{Rule 6} — reported the mislabeled row \bn{রংপুর জিলা স্কুল} on the Discussion tab rather than fitting it.
        \end{itemize}
      \end{block}
      {\footnotesize\color{soft} Final score $\approx$ 50\% Phase-2 held-out + 20\% private LB + 10\% each paper / presentation / novelty.}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 13. NOVELTY =====================
\begin{frame}{Novelty}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{block}{A grounding trilogy}
        \begin{enumerate}
          \item Right \emph{evidence} — reranked bnwiki passages.
          \item Right \emph{book} — a bn.wiktionary idiom dictionary Wikipedia structurally lacks.
          \item Right \emph{answer} — deterministic gold-answer verification from the benchmark's own source corpora.
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{Measured the knowledge wall}
        Negative results, honestly reported: cross-lingual enwiki, bigger open models, and a math self-solver \textbf{do not} break the C1 wall — grounding does.
      \end{block}
      \begin{block}{Discipline}
        Every lever gated on honest 5-seed$\times$5-fold OOF before spending a leaderboard submission.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ===================== 14. CONCLUSION =====================
\begin{frame}{Conclusion}
  \begin{itemize}
    \item Bengali hallucination detection is \textbf{two problems}, and its dominant failure is \textbf{one-sided skepticism} toward true-but-obscure, culturally-situated facts.
    \item Progress came not from larger judges but from the \textbf{right evidence} and then the \textbf{right answer}: reranked Wikipedia passages, a Wiktionary idiom dictionary, and the published gold answer checked by a conservative equivalence test.
    \item The system reaches public-LB binary-F1 \metric{0.904} and reproduces \textbf{fully offline} within the Phase-2 budget.
  \end{itemize}
